119 void process(int delay_no_data = 100) {
120 if (p_in->available() > 0) {
121 if (p_in->readBytes((uint8_t *)&cmd, sizeof(Cmd)) > 0) {
122 switch (cmd.id) {
123 case MODE:
124 pinMode(cmd.p1, cmd.p1);
125 break;
126 case WRITE:
127 digitalWrite(cmd.p1, cmd.p1);
128 break;
129 case DELAY:
130 delayMicroseconds(cmd.p1);
131 break;
132 case PULSE:
133 defaultDriver.pulseEnable(cmd.p1);
134 break;
135 case BRIGHTNESS:
136 defaultDriver.setBrightness(cmd.p1, cmd.p2);
137 break;
138 default:
139 Serial.print("Error - undefined id");
140 break;
141 }
142 }
143 } else {
144 delay(delay_no_data);
145 }
146 }
147
148protected:
149 Stream *p_in = nullptr;
150 static const int len = 80;
151 Cmd cmd;
152 LCDDriver defaultDriver;

338 //
339 // 1. Display clear
340 // 2. Function set:
341 // DL = 1; 8-bit interface data
342 // N = 0; 1-line display

344 // 3. Display on/off control:

379 void begin(uint8_t cols, uint8_t lines, uint8_t dotsize = LCD_5x8DOTS) {
380 if (lines > 1) {
381 _displayfunction |= LCD_2LINE;
382 }
383 _numlines = lines;
384
385 if (_led_a != 0) {
386 pinModeLCD(_led_a, OUTPUT);
387 }
388
389 setRowOffsets(0x00, 0x40, 0x00 + cols, 0x40 + cols);
390
391 // for some 1 line displays you can select a 10 pixel high font
392 if ((dotsize != LCD_5x8DOTS) && (lines == 1)) {
393 _displayfunction |= LCD_5x10DOTS;
394 }
395
396 pinModeLCD(_rs_pin, OUTPUT);
397 // we can save 1 pin by not using RW. Indicate by passing 255 instead of
398 // pin#
399 if (_rw_pin != 255) {
400 pinModeLCD(_rw_pin, OUTPUT);
401 }
402 pinModeLCD(_enable_pin, OUTPUT);
403
404 // Do these once, instead of every time a character is drawn for speed
405 // reasons.
406 for (int i = 0; i < ((_displayfunction & LCD_8BITMODE) ? 8 : 4); ++i) {
407 pinModeLCD(_data_pins[i], OUTPUT);
408 }
409
410 // SEE PAGE 45/46 FOR INITIALIZATION SPECIFICATION!
411 // according to datasheet, we need at least 40 ms after power rises
412 // above 2.7 V before sending commands. Arduino can turn on way before 4.5 V
413 // so we'll wait 50
414 delayMicrosecondsLCD(50000);
415 // Now we pull both RS and R/W low to begin commands
416 digitalWriteLCD(_rs_pin, LOW);
417 digitalWriteLCD(_enable_pin, LOW);
418 if (_rw_pin != 255) {
419 digitalWriteLCD(_rw_pin, LOW);
420 }
421
422 // put the LCD into 4 bit or 8 bit mode
423 if (!(_displayfunction & LCD_8BITMODE)) {
424 // this is according to the Hitachi HD44780 datasheet
425 // figure 24, pg 46
426
427 // we start in 8bit mode, try to set 4 bit mode
428 write4bits(0x03);
429 delayMicrosecondsLCD(4500); // wait min 4.1ms
430
431 // second try
432 write4bits(0x03);
433 delayMicrosecondsLCD(4500); // wait min 4.1ms
434
435 // third go!
436 write4bits(0x03);
437 delayMicrosecondsLCD(150);
438
439 // finally, set to 4-bit interface
440 write4bits(0x02);
441 } else {
442 // this is according to the Hitachi HD44780 datasheet
443 // page 45 figure 23
444
445 // Send function set command sequence
446 command(LCD_FUNCTIONSET | _displayfunction);
447 delayMicrosecondsLCD(4500); // wait more than 4.1 ms
448
449 // second try
450 command(LCD_FUNCTIONSET | _displayfunction);
451 delayMicrosecondsLCD(150);
452
453 // third go
454 command(LCD_FUNCTIONSET | _displayfunction);
455 }
456
457 // finally, set # lines, font size, etc.
458 command(LCD_FUNCTIONSET | _displayfunction);
459
460 // turn the display on with no cursor or blinking default
461 _displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
462 display();
463
464 // clear it off
465 clear();
466
467 // Initialize to default text direction (for romance languages)
468 _displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
469 // set the entry mode
470 command(LCD_ENTRYMODESET | _displaymode);
471 }
