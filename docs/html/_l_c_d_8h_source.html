120 void process(int delay_no_data = 100) {
121 if (p_in->available() > 0) {
122 if (p_in->readBytes((uint8_t *)&cmd, sizeof(Cmd)) > 0) {
123 switch (cmd.id) {
124 case MODE:
125 pinMode(cmd.p1, cmd.p1);
126 break;
127 case WRITE:
128 digitalWrite(cmd.p1, cmd.p1);
129 break;
130 case DELAY:
131 delayMicroseconds(cmd.p1);
132 break;
133 case PULSE:
134 defaultDriver.pulseEnable(cmd.p1);
135 break;
136 case BRIGHTNESS:
137 defaultDriver.setBrightness(cmd.p1, cmd.p2);
138 break;
139 default:
140 Serial.print("Error - undefined id");
141 break;
142 }
143 }
144 } else {
145 delay(delay_no_data);
146 }
147 }
148
149 protected:
150 Stream *p_in = nullptr;
151 static const int len = 80;
152 Cmd cmd;
153 LCDDriver defaultDriver;

348 // DL = 1; 8-bit interface data

386 void begin(uint8_t cols, uint8_t lines, uint8_t dotsize = LCD_5x8DOTS) {
387 if (lines > 1) {
388 _displayfunction |= LCD_2LINE;
389 }
390 _numlines = lines;
391
392 if (_led_a != 0) {
393 pinModeLCD(_led_a, OUTPUT);
394 }
395
396 setRowOffsets(0x00, 0x40, 0x00 + cols, 0x40 + cols);
397
398 // for some 1 line displays you can select a 10 pixel high font
399 if ((dotsize != LCD_5x8DOTS) && (lines == 1)) {
400 _displayfunction |= LCD_5x10DOTS;
401 }
402
403 pinModeLCD(_rs_pin, OUTPUT);
404 // we can save 1 pin by not using RW. Indicate by passing 255 instead of
405 // pin#
406 if (_rw_pin != 255) {
407 pinModeLCD(_rw_pin, OUTPUT);
408 }
409 pinModeLCD(_enable_pin, OUTPUT);
410
411 // Do these once, instead of every time a character is drawn for speed
412 // reasons.
413 for (int i = 0; i < ((_displayfunction & LCD_8BITMODE) ? 8 : 4); ++i) {
414 pinModeLCD(_data_pins[i], OUTPUT);
415 }
416
417 // SEE PAGE 45/46 FOR INITIALIZATION SPECIFICATION!
418 // according to datasheet, we need at least 40 ms after power rises
419 // above 2.7 V before sending commands. Arduino can turn on way before 4.5 V
420 // so we'll wait 50
421 delayMicrosecondsLCD(50000);
422 // Now we pull both RS and R/W low to begin commands
423 digitalWriteLCD(_rs_pin, LOW);
424 digitalWriteLCD(_enable_pin, LOW);
425 if (_rw_pin != 255) {
426 digitalWriteLCD(_rw_pin, LOW);
427 }
428
429 // put the LCD into 4 bit or 8 bit mode
430 if (!(_displayfunction & LCD_8BITMODE)) {
431 // this is according to the Hitachi HD44780 datasheet
432 // figure 24, pg 46
433
434 // we start in 8bit mode, try to set 4 bit mode
435 write4bits(0x03);
436 delayMicrosecondsLCD(4500); // wait min 4.1ms
437
438 // second try
439 write4bits(0x03);
440 delayMicrosecondsLCD(4500); // wait min 4.1ms
441
442 // third go!
443 write4bits(0x03);
444 delayMicrosecondsLCD(150);
445
446 // finally, set to 4-bit interface
447 write4bits(0x02);
448 } else {
449 // this is according to the Hitachi HD44780 datasheet
450 // page 45 figure 23
451
452 // Send function set command sequence
453 command(LCD_FUNCTIONSET | _displayfunction);
454 delayMicrosecondsLCD(4500); // wait more than 4.1 ms
455
456 // second try
457 command(LCD_FUNCTIONSET | _displayfunction);
458 delayMicrosecondsLCD(150);
459
460 // third go
461 command(LCD_FUNCTIONSET | _displayfunction);
462 }
463
464 // finally, set # lines, font size, etc.
465 command(LCD_FUNCTIONSET | _displayfunction);
466
467 // turn the display on with no cursor or blinking default
468 _displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
469 display();
470
471 // clear it off
472 clear();
473
474 // Initialize to default text direction (for romance languages)
475 _displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
476 // set the entry mode
477 command(LCD_ENTRYMODESET | _displaymode);
478 }

787 // -- setting initial values

795 void drawValue(int value, int maxValue) {
796 // -- calculate full (filled) character count
797 byte fullChars = (long)value * _numCols / maxValue;
798 // -- calculate partial character bar count
799 byte mod = ((long)value * _numCols * 5 / maxValue) % 5;
800
801 // -- if value does not change, do not draw anything
802 int normalizedValue = (int)fullChars * 5 + mod;
803 if (this->_prevValue != normalizedValue) {
804 // -- do not clear the display to eliminate flickers
805 _lcd->setCursor(_startX, _startY);
806
807 // -- write filled characters
808 for (byte i = 0; i < fullChars; i++) {
809#ifdef USE_BUILDIN_FILLED_CHAR
810 _lcd->write(
811 (byte)USE_BUILDIN_FILLED_CHAR); // -- use build in filled char
812#else
813 _lcd->write((byte)0);
814#endif
815 }
816
817 // -- write the partial character
818 if (mod > 0) {
819 _lcd->write(mod); // -- index the right partial character
820 ++fullChars;
821 }
822
823 // -- clear characters left over the previous draw
824 for (byte i = fullChars; i < this->_lastFullChars; i++) {
825 _lcd->write(' ');
826 }
827
828 // -- save cache
829 this->_lastFullChars = fullChars;
830 this->_prevValue = normalizedValue;
831 }
832 }
